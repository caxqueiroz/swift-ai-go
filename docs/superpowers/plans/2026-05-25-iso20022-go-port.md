# ISO 20022 Address Structuring Go Port Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a native Go library and CLI port of `Swift-SC/iso20022-address-structuring` with ONNX Runtime model inference and Go-native deterministic pipeline logic.

**Architecture:** The port is split into small packages: `normalize`, `readers`, `model`, `fuzzy`, `postcode`, `postprocess`, `resources`, `pipeline`, `output`, and `cmd/iso-run`. ONNX Runtime runs Transformer/country-head inference; Go implements tokenization, CRF decode/marginals, matching, scoring, and output formatting.

**Tech Stack:** Go 1.23+, `log/slog`, `encoding/csv`, `compress/zlib`, `encoding/json`, `github.com/yalue/onnxruntime_go`, `github.com/parquet-go/parquet-go`, `github.com/mozillazg/go-unidecode`, and `github.com/agnivade/levenshtein`.

## File Structure

Create and maintain these package boundaries:

- `cmd/iso-run/main.go`: CLI flag parsing, logging, pipeline execution, output writing.
- `internal/core/types.go`: shared result, match, sample, span, tag, and flag types.
- `internal/config/config.go`: default config values and CLI/env overrides.
- `internal/normalize/normalize.go`: transliteration and alias variation helpers.
- `internal/readers/readers.go`: text, CSV, TSV readers returning `core.AddressSample`.
- `internal/resources/resources.go`: compressed JSON, country, town, postcode, and parquet loaders.
- `internal/model/tokenizer.go`: character tokenizer.
- `internal/model/crf.go`: Viterbi decode, forward/backward marginals, log-sum-exp.
- `internal/model/span.go`: BIO tag grouping into spans.
- `internal/model/onnx.go`: ONNX Runtime session adapter.
- `internal/fuzzy/fuzzy.go`: candidate filtering, near matching, result types.
- `internal/postcode/postcode.go`: postcode regex matching and country-specific structures.
- `internal/postprocess/score.go`: town/country score formulas.
- `internal/postprocess/combinations.go`: country-town combination generation.
- `internal/postprocess/flags.go`: flag managers and relationship flags.
- `internal/postprocess/runner.go`: orchestration of scoring, flags, and final combinations.
- `internal/pipeline/pipeline.go`: batch orchestration equivalent to Python `AddressStructuringPipeline`.
- `internal/output/output.go`: JSON and human-readable CSV/TSV writers.
- `tools/convert-model/export_onnx.py`: one-time PyTorch/safetensors to ONNX + CRF JSON exporter.
- `tools/convert-model/README.md`: exact conversion command and artifact contract.

package normalize

import (
	"regexp"
	"sort"
	"strings"

	"github.com/mozillazg/go-unidecode"
)

var saintVariationRE = regexp.MustCompile(`\bSAINT([^\s\w]|\s)|\bST\.([^\s\w]|\s)|\bST([^\s\w]|\s)`)

func DecodeAndClean(s string) string {
	replacements := map[string]string{"@": "a", "`": "'", "–": "-"}
	out := unidecode.Unidecode(s)
	out = strings.ReplaceAll(out, "´", "'")
	out = strings.ReplaceAll(out, "‘", "'")
	out = strings.ReplaceAll(out, "’", "'")
	for old, next := range replacements {
		out = strings.ReplaceAll(out, old, next)
	}
	return out
}

func DuplicateIfSeparatorPresent(name string) []string {
	processed := strings.ReplaceAll(name, "-", "%TOKEN%")
	processed = strings.ReplaceAll(processed, " ", "%TOKEN%")
	values := unique([]string{
		strings.ReplaceAll(processed, "%TOKEN%", "-"),
		strings.ReplaceAll(processed, "%TOKEN%", " "),
	})
	return values
}

func DuplicateIfSaintInName(name string) []string {
	processed := saintVariationRE.ReplaceAllString(name, "%TOKEN%")
	return unique([]string{
		strings.ReplaceAll(processed, "%TOKEN%", "SAINT-"),
		strings.ReplaceAll(processed, "%TOKEN%", "ST. "),
		strings.ReplaceAll(processed, "%TOKEN%", "ST-"),
	})
}

func GenerateDuplicateAliases(name string) []string {
	values := []string{name}
	for _, saint := range DuplicateIfSaintInName(name) {
		values = append(values, DuplicateIfSeparatorPresent(saint)...)
	}
	return unique(values)
}

func unique(values []string) []string {
	seen := map[string]struct{}{}
	out := make([]string, 0, len(values))
	for _, value := range values {
		if _, ok := seen[value]; ok {
			continue
		}
		seen[value] = struct{}{}
		out = append(out, value)
	}
	sort.Strings(out)
	return out
}
```

package readers

import (
	"encoding/csv"
	"fmt"
	"io"
	"os"
	"strings"

	"github.com/tipmarket/swift-ai/internal/core"
)

const (
	DefaultAddressColumn               = "address"
	DefaultSuggestedCountryColumn      = "suggested_country"
	DefaultForceSuggestedCountryColumn = "force_suggested_country"
)

func ReadText(path string) ([]core.AddressSample, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return nil, fmt.Errorf("reading text file: %w", err)
	}
	lines := strings.Split(strings.ReplaceAll(string(data), "\r\n", "\n"), "\n")
	if len(lines) > 0 && lines[len(lines)-1] == "" {
		lines = lines[:len(lines)-1]
	}
	samples := make([]core.AddressSample, 0, len(lines))
	for _, line := range lines {
		samples = append(samples, core.AddressSample{Text: line})
	}
	return samples, nil
}

func ReadDelimited(path string, comma rune, addressColumn string) ([]core.AddressSample, error) {
	file, err := os.Open(path)
	if err != nil {
		return nil, fmt.Errorf("opening delimited file: %w", err)
	}
	defer file.Close()

	r := csv.NewReader(file)
	r.Comma = comma
	r.FieldsPerRecord = -1

	header, err := r.Read()
	if err != nil {
		return nil, fmt.Errorf("reading header: %w", err)
	}
	index := map[string]int{}
	for i, col := range header {
		index[col] = i
	}
	addrIdx, ok := index[addressColumn]
	if !ok {
		return nil, fmt.Errorf("Column '%s' not found in CSV file: %s", addressColumn, path)
	}
	suggestIdx, hasSuggest := index[DefaultSuggestedCountryColumn]
	forceIdx, hasForce := index[DefaultForceSuggestedCountryColumn]

	var samples []core.AddressSample
	for {
		record, err := r.Read()
		if err == io.EOF {
			break
		}
		if err != nil {
			return nil, fmt.Errorf("reading record: %w", err)
		}
		if addrIdx >= len(record) || record[addrIdx] == "" {
			continue
		}
		sample := core.AddressSample{Text: record[addrIdx]}
		if hasSuggest && suggestIdx < len(record) {
			value := strings.ToUpper(strings.TrimSpace(record[suggestIdx]))
			if value != "" {
				sample.SuggestedCountry = value
				sample.HasSuggestedCountry = true
			}
		}
		if hasForce && forceIdx < len(record) {
			sample.ForceSuggestedCountry = parseForceFlag(record[forceIdx])
		}
		samples = append(samples, sample)
	}
	return samples, nil
}

func parseForceFlag(value string) bool {
	switch strings.ToLower(strings.TrimSpace(value)) {
	case "true", "1", "yes", "y":
		return true
	default:
		return false
	}
}
```
